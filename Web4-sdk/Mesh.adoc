class WebRTCMesh {
  constructor(nodeId) {
    this.nodeId = nodeId;
    this.peers = new Map();
    this.connections = [];
    this.signaling = null; // external server required
  }
  async connect(signalServer) {
    this.signaling = signalServer;
    this.signaling.onmessage = async (msg) => {
      await this.handleSignal(msg);
    };
    this.signaling.send({
      type: "join",
      nodeId: this.nodeId
    });
  }
  async createPeer(targetId) {
    const pc = new RTCPeerConnection({
      iceServers: [{ urls: "stun:stun.l.google.com:19302" }]
    });
    pc.ondatachannel = (event) => {
      this.setupChannel(event.channel, targetId);
    };
    const channel = pc.createDataChannel("swarm");
    this.setupChannel(channel, targetId);
    const offer = await pc.createOffer();
    await pc.setLocalDescription(offer);
    this.signaling.send({
      type: "offer",
      to: targetId,
      from: this.nodeId,
      data: offer
    });
    this.peers.set(targetId, pc);
  }
  setupChannel(channel, peerId) {
    channel.onmessage = (e) => {
      const msg = JSON.parse(e.data);
      console.log("🌍 swarm msg:", msg);
    };
    this.connections.push({ peerId, channel });
  }
  broadcast(data) {
    for (const conn of this.connections) {
      conn.channel.send(JSON.stringify({
        from: this.nodeId,
        data
      }));
    }
  }
}

class GlobalAINode {
  constructor(brain, memory, mesh) {
    this.brain = brain;
    this.memory = memory;
    this.mesh = mesh;
    this.swarmContext = [];
  }
  async think(input, policy) {
    // local reasoning
    const local = await this.brain.think(input, await this.memory.getAll(), policy);
    // broadcast thought globally
    this.mesh.broadcast({
      type: "thought",
      input,
      output: local
    });
    return local;
  }
  receive(msg) {
    if (msg.data?.type === "thought") {
      this.swarmContext.push(msg.data.output);
    }
  }
}

class SwarmOrchestrator {
  constructor(node, mesh, chain) {
    this.node = node;
    this.mesh = mesh;
    this.chain = chain;
    this.consensusBuffer = [];
  }
  async process(input) {
    const result = await this.node.think(input, {
      mode: "global_swarm"
    });
    // evaluate importance
    const importance = result.length > 50;
    if (importance) {
      const hashBuffer = await this.chain.hashThought(result);
      const hashArray = Array.from(new Uint8Array(hashBuffer));
      const hash = hashArray.map(b => b.toString(16)).join("");
      await this.chain.storeThought(hash, {
        input,
        result
      });
    }
    this.consensusBuffer.push(result);
    return this.computeConsensus();
  }
  computeConsensus() {
    if (this.consensusBuffer.length < 3) return null;
    const best = this.consensusBuffer
      .sort((a,b) => b.length - a.length)[0];
    this.consensusBuffer = [];
    return best;
  }
}
